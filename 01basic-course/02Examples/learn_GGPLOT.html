Learn ggplot

  • elegance and versatility
  • “Grammar of Graphics”
  • ggplot(data = <DATA>, mapping = aes(<MAPPINGS>)) + <GEOM_FUNCTION>()

Create a dataframe

datf <- data.frame(
  xval = 1:5,
  yval = c(3,5,6,9,4),
  group = c("A","B","A","B","B")
)
datf
##   xval yval group
## 1    1    3     A
## 2    2    5     B
## 3    3    6     A
## 4    4    9     B
## 5    5    4     B

A Simple Example with ggplot()

library(ggplot2)
ggplot(datf, aes(x=xval,y=yval)) +
       geom_point()

Add options

Color points by group

ggplot(datf, aes(x=xval,y=yval)) +
       geom_point(aes(colour=group))

Use scale to control the mappings

ggplot(datf, aes(x=xval,y=yval)) +
  geom_point(aes(colour=group)) +
  scale_colour_manual(values=c("orange","forestgreen")) +
  scale_x_continuous(limits=c(0,8))

Add a line

ggplot(datf, aes(x=xval,y=yval)) +
       geom_line() +
       geom_point(aes(colour=group))

Modify theme, font, linewidth

theme_set(theme_bw(base_size = 16, base_family = "Times"))
ggplot(datf, aes(x=xval,y=yval)) +
      geom_line(linewidth=1.5) +
      geom_point(aes(colour=group))
